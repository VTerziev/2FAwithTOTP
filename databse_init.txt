CREATE TABLE users(
	id BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
	user VARCHAR(32) NOT NULL UNIQUE,
	pass VARCHAR(128) NOT NULL,
	2fa BOOLEAN NOT NULL,
	shared_key VARCHAR(512) NOT NULL
) CHARSET=ascii, COLLATE=ascii_general_ci;


CREATE TABLE apps(
	id BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY,
	name VARCHAR(32) NOT NULL UNIQUE,
	shared_key VARCHAR(128) NOT NULL
) CHARSET=ascii, COLLATE=ascii_general_ci;
